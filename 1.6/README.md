# Example Presta
```
const Api = require('prestashop-sdk/1.6')

// type is simply the ec-name name of the ec module
var myPresta = new Api({
      apiKey: 'kkkkkkkkkkkkk',
      storeUrl: 'https://yourdomain.tld/presta',
      options: {
        debug: true
      }
  })

// myPresta === CompletApi Model
myPresta.get('orders').then(function(response) {
    console.log(response);
}).catch(function(errors) {
    console.log(errors);
});

//Or

myPresta.get('orders', {id: 1}).then(function(response) {
    console.log(response);
}).catch(function(errors) {
    console.log(errors);
});

//And you can even use the filter querystrings

myPresta.get('orders', {
    'display': 'full',
    'filter[id]': 1
}).then(function(response) {
    console.log(response);
}).catch(function(errors) {
    console.log(errors);
});

//You can add/edit too.

myPresta.get('products', {
    id: 1
}).then(function(response) {
    response.prestashop.product.price = '1.99';
    return myPresta.edit('products', response, {
        id: 1
    });
}).catch(function(errors) {
    console.log(errors);
});
```
